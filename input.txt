fn sqrt(n: double) -> double {
    let l: double = 0.0;
    let r: double = n;
    let mid: double;
    let mid2: double;
    while r-l > 0.000001 {
        mid = (l+r) / 2.0;
        mid2 = mid*mid;
        if (mid2 == n) {
            return mid;
        }
        if (mid2 < n) {
            l = mid;
        }
        else {
            r = mid;
        }
    }
    return mid;
}
